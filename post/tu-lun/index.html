1.21.21.2 概念

有/无向图：如果给图的每条边规定一个方向，那么得到的图称为有向图。在有向图中，与一个节点相关联的边有出边和入边之分。相反，边没有方向的图称为无向图。

度(DegreeDegreeDegree)：一个顶点的度是指与该顶点相关联的边的条数，顶点vvv的度记作d(v)d(v)d(v)

入度(In−degreeIn-degreeIn−degree)和出度(Out−degreeOut-degreeOut−degree)：对于有向图来说，一个顶点的度可细分为入度和出度。一个顶点的入度是指与其关联的各边之中，以其为终点的边数；出度则是相对的概念，指以该顶点为起点的边数。

自环(LoopLoopLoop)：若一条边的两个顶点为同一顶点，则此边称作自环。

路径(PathPathPath)：从uuu到vvv的一条路径是指一个序列v0,e1,v1,e2,v2,⋯ ,ek,vkv_0,e_1,v_1,e_2,v_2,\cdots ,e_k,v_kv0​,e1​,v1​,e2​,v2​,⋯,ek​,vk​，其中eie_iei​的顶点为viv_ivi​及vi−1v_{i-1}vi−1​，kkk称作路径的长度。如果它的起止顶点相同，该路径是“闭”的，反之，则称为“开”的。一条路径称为一简单路径(simplepathsimple pathsimplepath)，如果路径中除起始与终止顶点可以重合外，所有顶点两两不等。

桥/割边(BridgeBridgeBridge)：若去掉一条边，便会使得整个图不连通，该边称为桥，见5.35.35.3。

割点：若去掉一个点以及与它关联的边后，便会使得整个图不连通，该点称为割点，见5.35.35.3。

vector<int> topsort() {
	priority_queue<int, vector<int>, greater<int> > doing;
	for (int i = 1; i <= n; i++) {
		if (ind[i] == 0) {
			doing.push(i);
		}
	}
	int u;
	vector<int> ans;
	int times = 0;
	while (!doing.empty()) {
		++times;
		u = doing.top();
		ans.push_back(u);
		doing.pop();
		for (int i = 0; i < G[u].size(); i++) {
			dp[G[u][i].first] = max(dp[G[u][i].first], dp[u] + G[u][i].second);
			ind[G[u][i].first]--;
			if (ind[G[u][i].first] == 0) {
				ind[G[u][i].first]--;
				doing.push(G[u][i].first);
			}
		}
	}
	return times == n ? ans : vector<int>();
}

3.43.43.4 1449 -- 【MST练习】最小生成树1250

Description

给你一个具有n个点，m条边的无向连通图，求其最小生成树的权值和。

#include <cstdio>
#include <algorithm>
using namespace std;
const int MAXN = 301;
const int MAXM = 90001;
struct node {
	int u, v, w;
} road[MAXM];
int ans;
bool cmp(node x, node y) {
	if (x.w == y.w) {
		if (x.u == x.u)
			return x.v < y.v;
		else
			return x.u < y.u;
	}
	else return x.w < y.w;
}
int fa[MAXN];
void makeset(int n) {
	for (int i = 1; i  <= n; i++) {
		fa[i] = i;
	}
	return;
}
int findroot(int x) {
	if (fa[x] != x) return fa[x] = findroot(fa[x]);
	else return x;
}
void unionset(int u, int v, int w) {
	int x = findroot(u), y = findroot(v);
	if (x == y) return;
	fa[x] = y;
	ans += w;
	return;
}
int main() {
	int n, m;
	scanf("%d %d", &n, &m);
	for (int i = 0; i < m; i++) {
        scanf("%d %d %d", &road[i].u, &road[i].v, &road[i].w);
	}
	makeset(n);
	stable_sort(road, road + m, cmp);
	for (int i = 0; i < m; i++) {
		unionset(road[i].u, road[i].v, road[i].w);
	}
	printf("%d", ans);
	return 0;
}

5.45.45.4 1572 -- 【图的连通性】消息的传递2269

Description

我们的郭嘉大大在曹操这过得逍遥自在，但是有一天曹操给了他一个任务，在建邺城内有N（<=1000）个袁绍的奸细，将他们从1到N进行编号，同时他们之间存在一种传递关系，即若C[i，j]=1，则奸细i能将消息直接传递给奸细j。

#include <cstdio>
#include <algorithm>
#include <vector>
#include <queue>
#include <set>
using namespace std;
const int MAXN = 102;
vector<pair<int, long long> > G[MAXN];
set<int> final_path;
vector<int> lastu[MAXN];
int ind[MAXN];
int n;
long long dp[MAXN];
pair<long long, int> topsort() {
	queue<int> doing;
	for (int i = 1; i <= n; i++) {
		if (ind[i] == 0) {
			doing.push(i);
		}
	}
	int u;
	while (!doing.empty()) {
		u = doing.front();
		doing.pop();
		for (int i = 0; i < G[u].size(); i++) {
			if (dp[G[u][i].first] < dp[u] + G[u][i].second) {
				dp[G[u][i].first] = dp[u] + G[u][i].second;
				lastu[G[u][i].first].clear();
				lastu[G[u][i].first].push_back(u);
			}
			else if (dp[G[u][i].first] == dp[u] + G[u][i].second) {
				lastu[G[u][i].first].push_back(u);
			}
			ind[G[u][i].first]--;
			if (ind[G[u][i].first] == 0) {
				ind[G[u][i].first]--;
				doing.push(G[u][i].first);
			}
		}
	}
	return make_pair(dp[n], n);
}
void FindWay(int u) {
	if (!lastu[u].empty()) {
		for (int i = 0; i < lastu[u].size(); i++) {
			FindWay(lastu[u][i]);
		}
	}
	final_path.insert(u);
}

		int u, v;
		long long w;

		ind[v]++;
	}
