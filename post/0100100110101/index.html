矩阵快速幂的实现

这时有人就要问了：“YJYJ！怎么实现呀”

I have a matrix, I have a quick pow. Oh quick matrix pow!

我们先调用出古老的快速幂：

const int mod = 114514;
unsigned long long qkpow(unsigned long long a, unsigned long long b) {
    unsigned long long ans = 1;
    while (b) {
        if (b bitand 1) {
            ans = ans * a % mod;
        }
        a = a * a % mod;
        b >>= 1;
    }
    return ans;
}

我们发现只需要把unsigned long long a换为矩阵a即可。

我们可以写一个矩阵struct：

const int MAXN = 『矩阵的大小』
struct matrix {
	long long arr[MAXN][MAXN];
	int n, m;
};

再写一个矩阵乘法的函数

为了运算A * B % mod

代码如下：

const int mod = 114514;
matrix mul(matrix a, matrix b) {
	matrix c;
	memset(c.arr, 0, sizeof c.arr);
	c.n = a.n;
	c.m = b.m;
	for (int i = 0; i < c.n; i++) {
		for (int j = 0; j < c.m; j++) {
			c.arr[i][j] = 0;
			for (int k = 0; k < a.m; k++) {
				c.arr[i][j] += a.arr[i][k] * b.arr[k][j];
				c.arr[i][j] = c.arr[i][j] % mod;
			}
		}
	}
	return c;
}

而ans的初始值可以用单位矩阵来代替

于是我们就可以运用上面两段代码来写出矩阵快速幂了：

matrix qkpow(matrix a, int b) {
	matrix ans;
	ans.arr[0][0] = 1;
	ans.arr[1][1] = 1;
	ans.arr[0][1] = 0;
	ans.arr[1][0] = 0;
	ans.n = 2;
	ans.m = 2;
	while (b) {
		if (b & 1) {
			ans = mul(ans, a);
		}
		a = mul(a, a);
		b >>= 1;
	}
	return ans;
}
